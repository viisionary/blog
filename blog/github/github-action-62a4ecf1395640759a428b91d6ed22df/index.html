配备ubuntu环境

  1. npm ci 安装包
  2. 运行build指令
  3. 编辑好的代码在spider-utils
  4. 发布
name: Node.js Package

on:
  push:
    branches:
      - master
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - uses: actions/setup-node@v2
        with:
          node-version: 14
          registry-url: https://registry.npmjs.org
      - run: npm ci
      - run: npm run build
      - run: |
          cd spider-utils
          npm publish          
        env:
          NODE_AUTH_TOKEN: ${{secrets.NPM_TOKEN}}
