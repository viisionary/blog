const Person = {
	name: "Joey",
	age: 25
};
const entries = Object.entries(Person);
console.log(entries);
const restore = Object.fromEntries(entries);
console.log(restore);
// array
const persons = [[[Person]], { ...Person, name: "peter" }];
console.log("flat", persons.flat(), persons.flat().flat());
console.log(
	"flatMap",
	persons.flatMap((x) => {
		return [{ name: "name is" + x.name }];
	})
);
